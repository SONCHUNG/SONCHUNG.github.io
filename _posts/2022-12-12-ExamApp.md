package ex2;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.PrintStream;
import java.util.Scanner;

public class ExamApp {

	public static void main(String[] args) throws IOException {
		int answer = 1;
//		int kor1 = 0, kor2 = 0, kor3 = 0;
		int[] kors = new int[3];
		int total;
		double avg;

		LABEL: // Break LABEL
		while (answer == 1) {
			// 메인메뉴
			int menu;
			{
				Scanner scan = new Scanner(System.in);
				System.out.println("┌──────────────────────┐");
				System.out.println("│        메인메뉴        │");
				System.out.println("└──────────────────────┘");
				System.out.println("       1. 성적입력");
				System.out.println("       2. 파일입력");
				System.out.println("       3. 성적출력");
				System.out.println("       4. 파일저장");
				System.out.println("       5. 종   료");
				System.out.println(" ");
				System.out.print("   선택(1~5) > ");
				menu = scan.nextInt();
			}
			// 성적입력
			// System.out.println("1번 검사 전");
			switch (menu) {
			case 1: {
				Scanner scan = new Scanner(System.in);

				System.out.println("┌──────────────────────┐");
				System.out.println("│       \"성적입력\"       │");
				System.out.println("└──────────────────────┘");
				// kor1------------------------
				for (int i = 0; i < 3; i++) {
					int kor;

					do {
						System.out.printf("kor%d", i + 1);
						kor = scan.nextInt();

						if (kor < 0 || kor > 100)
							System.out.println("값이 유효하지 않습니다. (kor = 0~100)");

					} while (kor < 0 || kor > 100);

					kors[i] = kor;
				}
			}
				break;

			// 파일입력------------------------------------------
			// System.out.println("2번 검사 전");
			case 2: {
				FileInputStream fis = new FileInputStream("res/data.csv"); /* 컨트롤+쉬프트+O */
				Scanner scan = new Scanner(fis);

				String line = scan.nextLine();
				String[] tokens = line.split(",");

				for (int i = 0; i < 3; i++)
					kors[i] = Integer.parseInt(tokens[i]);

				scan.close();
				fis.close(); // 닫아주고 시작
			}
				break;
			// ------------------------------------------
			// 성적출력
			// System.out.println("3번 검사 전");
			case 3:

				System.out.println("┌──────────────────────┐");
				System.out.println("│       \"성적출력\"       │");
				System.out.println("└──────────────────────┘");

				for (int j = 0; j < 3; j++) {

					total = 0;

					for (int i = 0; i < 3; i++)
						total += kors[i];
					avg = total / 3.0;
					System.out.printf("-----------< %d >-----------\n", j + 1);
					for (int i = 0; i < 3; i++)
						System.out.printf(" 국어 %d : %3d\n", i + 1, kors[i]);

					System.out.printf(" 총  점 : %3d\n", total);
					System.out.printf(" 평  균 : %6.2f\n", avg);
					System.out.println("──────────────────────────");
				}
				break;
			// 파일저장 블록
			// System.out.println("4번 검사 전");
			case 4: {
				FileOutputStream fos = new FileOutputStream("res/data.csv");
				PrintStream out = new PrintStream(fos); /* 해당 코드 입력으로 파일출력스트림에서도 프린트스트림 응용객체 사용 가능해짐 */

				for (int i = 0; i < 3; i++) {
					out.printf("%d", kors[i]);

					if (i == 2)
						out.print("\n");
					else
						out.print(",");
				}
				fos.close();

				System.out.println("**********작업완료**********");
			}
				break;
			// 종료---------------------------
			// 계속하시겠습니까? (계속:1/종료:0)
			// if(answer == 0);
			// System.out.println("5번 검사 전");
			case 5: {
				Scanner scan = new Scanner(System.in);
				System.out.println("계속하시겠습니까? (계속:1/종료:0)");
				System.out.print("answer:");
				answer = scan.nextInt();
				scan.nextLine();
				// if (answer == 0)
				// break;
			}
			default: {
				System.out.println("1~5까지만 입력할 수 있습니다.");
				System.out.println("치명적인 오류로 프로그램을 종료합니다.");
				break LABEL; // LABEL을 사용함으로써 While문을 벗어날 수 있음.
			}

			}
		}
		System.out.println("Bye~");
	}
}
